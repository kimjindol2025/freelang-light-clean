// 1. 자동완성
const suggestions = interactiveMode.getAutocompletesuggestions('sum');
// → ['sum', 'sumAll', 'summation', ...]

// 2. 대화형 모드
interactiveMode.formatProposal('배열 합산', 'sum', 0.92, '배열 집계 패턴');
// → 신뢰도 바 포함한 포맷팅된 제안 표시

// 3. 피드백 기록

// → 피드백 저장 + 패턴 학습 업데이트

// 4. 대시보드
